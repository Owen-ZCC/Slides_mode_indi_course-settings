## 功能模块

<module id="1" name="口语对话">

  <feature id="1.1" name="人机自由对话" role="学生">
    <description>支持由场景内配置的对话角色与学生进行语音对话</description>
    <requirements>
      <requirement priority="high">
        对话内容显示为语音和英文文本
      </requirement>
      <requirement priority="high">
        记录口语对话内容、轮次、学生流畅度、发音情况
      </requirement>
      <requirement priority="high">
        对话过程中根据学生表现动态调整AI输出难度
      </requirement>
      <requirement priority="medium">
        对话过程中进行发音或内容的即时鼓励性反馈
      </requirement>
      <requirement priority="medium">
        识别录入语音的升降调、重音位置
      </requirement>
    </requirements>
  </feature>

  <feature id="1.2" name="教师看板" role="教师">
    <description>教师实时监控和管理学生对话的控制台</description>
    <requirements>
      <requirement priority="high">
        任务过程中实时汇总查看学生对话轮次、卡顿频率、错误类型等数据
      </requirement>
      <requirement priority="high">
        点击回放单个学生的对话记录
      </requirement>
      <requirement priority="medium">
        系统自动标记需重点辅导的学生，支持教师点击查看或回听
      </requirement>
      <requirement priority="medium">
        教师管理当前学生会话（结束、重新开始等）
      </requirement>
    </requirements>
  </feature>

</module>

<module id="2" name="课程创建与实施">

  <feature id="2.1" name="课程创建" role="教师">
    <description>教师创建和编辑网页版课件的工具</description>
    <requirements>
      <requirement priority="high" type="file-upload">
        支持自本地上传 .pptx 文件并解析为网页版课件
      </requirement>
      <requirement priority="high" type="content-edit">
        在网页版课件中新增页面，支持添加单选题、多选题、问答题
      </requirement>
      <requirement priority="medium" type="embed">
        支持添加网页（H5页面）至网页版课件
      </requirement>
      <requirement priority="high" type="ai-integration">
        调用AI应用中心，添加AI应用使用界面至网页版课件
      </requirement>
      <requirement priority="medium" type="ai-integration">
        调用AI创作中心，添加AI应用编辑界面至网页版课件
      </requirement>
      <requirement priority="low" type="media">
        自本地上传视频文件，或调用哔哩哔哩网页检索返回视频，添加至网页版课件
      </requirement>
    </requirements>
  </feature>

  <feature id="2.2" name="课程实施" role="教师/学生">
    <description>课程发布、学生参与和数据统计功能</description>
    <requirements>
      <requirement priority="high" type="access">
        自动生成课程识别码，支持学生角色输入课程识别码快捷登录
      </requirement>
      <requirement priority="high" type="analytics">
        统计学生端选择题、问答题、AI应用使用、AI应用编辑、网页的提交统计结果和回答详情
      </requirement>
      <requirement priority="medium" type="ai-assistant">
        对课程内容、工具回答结果进行AI对话，支持提问建议选项
      </requirement>
    </requirements>
  </feature>

</module>

<module id="3" name="用户系统对接">

  <feature id="3.1" name="用户登录" role="教师/学生">
    <description>与坪山区教育平台的用户系统集成</description>
    <requirements>
      <requirement priority="critical" type="integration">
        对接坪山区教育平台用户登录系统，实现用户身份验证及跳转
      </requirement>
    </requirements>
  </feature>

</module>

## 技术要点摘要

<technical-summary>
  <ai-capabilities>
    <capability>语音识别（ASR）- 支持升降调、重音位置识别</capability>
    <capability>语音合成（TTS）- AI角色语音输出</capability>
    <capability>自适应难度调整 - 根据学生表现动态调整</capability>
    <capability>发音评估 - 流畅度、准确度分析</capability>
    <capability>AI对话助手 - 课程内容问答</capability>
  </ai-capabilities>

  <integrations>
    <integration type="sso">坪山区教育平台用户登录系统</integration>
    <integration type="external">哔哩哔哩视频检索API</integration>
    <integration type="internal">AI应用中心</integration>
    <integration type="internal">AI创作中心</integration>
  </integrations>

  <file-formats>
    <format>.pptx - 课件上传解析</format>
    <format>H5页面 - 网页嵌入</format>
    <format>视频文件 - 本地上传</format>
  </file-formats>
</technical-summary>
